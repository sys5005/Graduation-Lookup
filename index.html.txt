    <button onclick="searchSeat()">بحث</button>
    <div id="result"></div>

    <script>
        function searchSeat() {
            var id = document.getElementById("employeeId").value;
            var url = "https://script.google.com/macros/s/AKfycbx_vSYp12PxVDu-o6MQ2J5ga-iZjb7XL_yOuCu5kjK3Pn98UN6BAAo_ganPRk1rnegzPg/exec" + "?id=" + id;

            fetch(url)
            .then(response => response.json())
            .then(data => {
                if (data.name) {
                    document.getElementById("result").innerHTML = 
                        "<strong>الاسم:</strong> " + data.name + "<br>" +
                        "<strong>رقم المقعد:</strong> " + data.seat;
                } else {
                    document.getElementById("result").innerHTML = "لم يتم العثور على الرقم الوظيفي.";
                }
            })
            .catch(error => console.error("Error:", error));
        }
    </script>
